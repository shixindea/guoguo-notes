```html
<a-date-picker
               v-model:value="model.validDate.deadline"
               format="YYYY-MM-DD"
               valueFormat="YYYY-MM-DD"
               :disabled="!!theId"
               :disabled-date="disabledDate"
               />
```

```js
// 限制选择日期 start
const disabledDate = (current: Dayjs) => {
    if (dayjs().endOf("day").format("YYYY-MM-DD") == current.format("YYYY-MM-DD")) {
        return false
    } else {
        return current < dayjs().endOf("day") 
    }
};
// 限制选择日期 end
```
