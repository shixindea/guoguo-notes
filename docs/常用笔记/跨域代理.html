и§ЈеҶіејҖеҸ‘зҺҜеўғзҡ„и·Ёеҹҹй—®йўҳ

ејҖеҸ‘зҺҜеўғзҡ„и·Ёеҹҹ

ејҖеҸ‘зҺҜеўғзҡ„и·ЁеҹҹпјҢд№ҹе°ұжҳҜеңЁ**vue-cliи„ҡжүӢжһ¶зҺҜеўғдёӢејҖеҸ‘еҗҜеҠЁжңҚеҠЎж—¶пјҢжҲ‘д»¬и®ҝй—®жҺҘеҸЈжүҖйҒҮеҲ°зҡ„и·Ёеҹҹй—®йўҳпјҢvue-cliдёәжҲ‘д»¬еңЁжң¬ең°ејҖеҗҜдәҶдёҖдёӘжңҚеҠЎ,еҸҜд»ҘйҖҡиҝҮиҝҷдёӘжңҚеҠЎеё®жҲ‘д»¬д»ЈзҗҶиҜ·жұӮ**,и§ЈеҶіи·Ёеҹҹй—®йўҳ

иҝҷе°ұжҳҜvue-cliй…ҚзҪ®webpackзҡ„еҸҚеҗ‘д»ЈзҗҶ

![image-20200811022013103](G:/web зҷҫеәҰзҪ‘зӣҳ/03гҖҒз¬¬дёүз«  HR-saasдёӯеҸ°з®ЎзҗҶйЎ№зӣ®/иө„жәҗ/и®Ід№ү/assets/image-20200811022013103.png)

йҮҮз”Ёvue-cliзҡ„д»ЈзҗҶй…ҚзҪ®

vue-cliзҡ„й…ҚзҪ®ж–Үд»¶еҚі**vue.config.js**,иҝҷйҮҢжңүжҲ‘д»¬йңҖиҰҒзҡ„ д»ЈзҗҶйҖүйЎ№

module.exports = {
  devServer: {
   // д»ЈзҗҶй…ҚзҪ®
    proxy: {
        // иҝҷйҮҢзҡ„api иЎЁзӨәеҰӮжһңжҲ‘д»¬зҡ„иҜ·жұӮең°еқҖжңү/apiзҡ„ж—¶еҖҷ,е°ұеҮәи§ҰеҸ‘д»ЈзҗҶжңәеҲ¶
        // localhost:8888/api/abc  => д»ЈзҗҶз»ҷеҸҰдёҖдёӘжңҚеҠЎеҷЁ
        // жң¬ең°зҡ„еүҚз«Ҝ  =гҖӢ жң¬ең°зҡ„еҗҺз«Ҝ  =гҖӢ д»ЈзҗҶжҲ‘д»¬еҗ‘еҸҰдёҖдёӘжңҚеҠЎеҷЁеҸ‘иҜ·жұӮ пјҲиЎҢеҫ—йҖҡпјү
        // жң¬ең°зҡ„еүҚз«Ҝ  =гҖӢ еҸҰеӨ–дёҖдёӘжңҚеҠЎеҷЁеҸ‘иҜ·жұӮ пјҲи·Ёеҹҹ иЎҢдёҚйҖҡпјү
        '/api': {
        target: 'www.baidu.com', // жҲ‘д»¬иҰҒд»ЈзҗҶзҡ„ең°еқҖ
        changeOrigin: true, // жҳҜеҗҰи·Ёеҹҹ йңҖиҰҒи®ҫзҪ®жӯӨеҖјдёәtrue жүҚеҸҜд»Ҙи®©жң¬ең°жңҚеҠЎд»ЈзҗҶжҲ‘д»¬еҸ‘еҮәиҜ·жұӮ
         // и·Ҝеҫ„йҮҚеҶҷ
        pathRewrite: {
            // йҮҚж–°и·Ҝз”ұ  localhost:8888/api/login  => www.baidu.com/api/login
            '^/api': '' // еҒҮи®ҫжҲ‘д»¬жғіжҠҠ localhost:8888/api/login еҸҳжҲҗwww.baidu.com/login е°ұйңҖиҰҒиҝҷд№ҲеҒҡ 
        }
      },
    }
  }
}

д»ҘдёҠе°ұжҳҜжҲ‘д»¬еңЁvue-cliйЎ№зӣ®дёӯй…ҚзҪ®зҡ„д»ЈзҗҶи®ҫзҪ®

жҺҘдёӢжқҘ,жҲ‘д»¬еңЁд»Јз Ғдёӯе°ҶиҰҒд»ЈзҗҶзҡ„еҗҺз«Ҝең°еқҖеҸҳжҲҗ еҗҺз«ҜжҺҘеҸЈең°еқҖ

 // д»ЈзҗҶи·Ёеҹҹзҡ„й…ҚзҪ®
    proxy: {
      // еҪ“жҲ‘д»¬зҡ„жң¬ең°зҡ„иҜ·жұӮ жңү/apiзҡ„ж—¶еҖҷпјҢе°ұдјҡд»ЈзҗҶжҲ‘д»¬зҡ„иҜ·жұӮең°еқҖеҗ‘еҸҰеӨ–дёҖдёӘжңҚеҠЎеҷЁеҸ‘еҮәиҜ·жұӮ
      '/api': {
        target: 'http://ihrm-java.itheima.net/', // и·ЁеҹҹиҜ·жұӮзҡ„ең°еқҖ
        changeOrigin: true // еҸӘжңүиҝҷдёӘеҖјдёәtrueзҡ„жғ…еҶөдёӢ жүҚиЎЁзӨәејҖеҗҜи·Ёеҹҹ
      }
    }

жң¬иҠӮжіЁж„ҸпјҡжҲ‘д»¬е№¶жІЎжңүиҝӣиЎҢ**pathRewrite,еӣ дёәеҗҺз«ҜжҺҘеҸЈе°ұжҳҜihrm-java.itheima.net/api**иҝҷз§Қж јејҸ,жүҖд»ҘдёҚйңҖиҰҒйҮҚеҶҷ

**vue.config.js**зҡ„ж”№еҠЁеҰӮжһңиҰҒз”ҹж•Ҳ,йңҖиҰҒиҝӣиЎҢйҮҚеҗҜжңҚеҠЎ

еҗҢж—¶пјҢиҝҳйңҖиҰҒжіЁж„Ҹзҡ„жҳҜпјҢжҲ‘д»¬еҗҢж—¶йңҖиҰҒжіЁйҮҠжҺү mockзҡ„еҠ иҪҪпјҢеӣ дёәmock-serverдјҡеҜјиҮҙд»ЈзҗҶжңҚеҠЎзҡ„ејӮеёё

// before: require('./mock/mock-server.js'),  // жіЁйҮҠmock-serverеҠ иҪҪ

з”ҹдә§зҺҜеўғзҡ„и·Ёеҹҹ

з”ҹдә§зҺҜеўғиЎЁзӨәжҲ‘д»¬е·Із»ҸејҖеҸ‘е®ҢжҲҗйЎ№зӣ®пјҢе°ҶйЎ№зӣ®йғЁзҪІеҲ°дәҶжңҚеҠЎеҷЁдёҠ,иҝҷж—¶е·Із»ҸжІЎжңүдәҶvue-cliи„ҡжүӢжһ¶зҡ„**иҫ…еҠ©дәҶпјҢжҲ‘д»¬еҸӘжҳҜжҠҠжү“еҢ…еҘҪзҡ„html+js+cssдәӨд»ҳиҝҗз»ҙдәәе‘ҳпјҢж”ҫеҲ°NginxжңҚеҠЎеҷЁиҖҢе·І,жүҖд»ҘжӯӨж—¶йңҖиҰҒеҖҹеҠ©Nginx**зҡ„еҸҚеҗ‘д»ЈзҗҶжқҘиҝӣиЎҢ

server{
    # зӣ‘еҗ¬9099з«ҜеҸЈ
    listen 9099;
    # жң¬ең°зҡ„еҹҹеҗҚжҳҜlocalhost
    server_name localhost;
    #еҮЎжҳҜlocalhost:9099/apiиҝҷдёӘж ·еӯҗзҡ„пјҢйғҪиҪ¬еҸ‘еҲ°зңҹжӯЈзҡ„жңҚеҠЎз«Ҝең°еқҖhttp://baidu.com
    location ^~ /api {
        proxy_pass http://baidu.com;
    }    
}

жіЁж„Ҹ:иҝҷйҮҢзҡ„ж“ҚдҪңдёҖиҲ¬з”ұиҝҗз»ҙдәәе‘ҳе®ҢжҲҗ,йңҖиҰҒеүҚз«ҜиҝӣиЎҢж“ҚдҪң,иҝҷйҮҢжҲ‘д»¬иҝӣиЎҢдёҖдёӢз®ҖеҚ•дәҶи§Ј

жӣҙеӨҡжӯЈеҗ‘д»ЈзҗҶе’ҢеҸҚеҗ‘д»ЈзҗҶзҹҘиҜҶ,иҜ·йҳ…иҜ»иҝҷзҜҮж–Үз« NginxеҸҚеҗ‘д»ЈзҗҶ

жҸҗдәӨд»Јз Ғ

жң¬иҠӮжіЁж„ҸпјҡжҲ‘д»¬е№¶жІЎжңүиҝӣиЎҢ**pathRewrite,еӣ дёәеҗҺз«ҜжҺҘеҸЈе°ұжҳҜihrm-java.itheima.net/api**иҝҷз§Қж јејҸ,жүҖд»ҘдёҚйңҖиҰҒйҮҚеҶҷ

жң¬иҠӮд»»еҠЎпјҡ й…ҚзҪ®vue-cliзҡ„еҸҚеҗ‘д»ЈзҗҶпјҢе®һзҺ°еҗҺз«ҜжҺҘеҸЈзҡ„и·Ёеҹҹи®ҝй—®
